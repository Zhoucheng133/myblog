# 遍历列表

```js
// App.js

const list=[
  { id: 1, name: "张三" },
  { id: 2, name: "李四" },
  { id: 3, name: "王五" },
]

function App() {
  // 注意一定要为每个遍历的DOM添加上key的值
  return (
    <div>
      { list.map(item => <li key={ item.id }> { item.id } : {item.name} </li>) }
    </div>
  );
}

export default App;

```
