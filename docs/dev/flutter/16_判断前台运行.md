# 判断前台运行

注意，一般来说用在移动端上

```dart
// 注意添加 with WidgetsBindingObserver
class _MainAppState extends State<MainApp> with WidgetsBindingObserver{
  @override
  void didChangeAppLifecycleState(AppLifecycleState state) {  // [!code ++]
    if (state == AppLifecycleState.resumed) { // [!code ++]
      // 应用进入前台 // [!code ++]
    } else {  // [!code ++]
      // 应用进入后台 // [!code ++]
    } // [!code ++]
  }

  @override
  void dispose() {
    WidgetsBinding.instance.removeObserver(this); // [!code ++]
    super.dispose();
  }

  @override
  void initState() {
    super.initState();
    // 注意在initState中绑定
    WidgetsBinding.instance.addObserver(this);  // [!code ++]
  }
}
```
